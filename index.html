        // --- LOCAL STORAGE FUNCTIONS ---

        const COURSES_STORAGE_KEY = 'learningAppCourses';

        function saveCoursesToStorage() {

            const coursesArray = Array.from(loadedCourses.entries());

            localStorage.setItem(COURSES_STORAGE_KEY, JSON.stringify(coursesArray));

        }

        function loadCoursesFromStorage() {

            const storedCourses = localStorage.getItem(COURSES_STORAGE_KEY);

            if (storedCourses) {

                const coursesArray = JSON.parse(storedCourses);

                loadedCourses = new Map(coursesArray);

                renderLoadedCourses();

            }

        }

        // Parser XML

        function parseXML(rawText) {

            const parser = new DOMParser();

            const xmlDoc = parser.parseFromString(rawText, "text/xml");

           

            if (xmlDoc.getElementsByTagName("parsererror").length > 0) return null;

           

            const pathwayNode = xmlDoc.getElementsByTagName("pathway")[0];

            if (!pathwayNode) return null;

            const pathwayData = {

                title: pathwayNode.getAttribute("title") || "Corso",

                cranes: []

            };

            const tileNodes = Array.from(pathwayNode.getElementsByTagName("TILE"));

            tileNodes.forEach((tileNode, tileIndex) => {

                const craneId = `crane-${tileIndex + 1}`;

                const crane = {

                    id: craneId,

                    title: tileNode.querySelector("title")?.textContent.trim() || `Capitolo ${tileIndex + 1}`,

                    tiles: []

                };

                const orbNodes = Array.from(tileNode.getElementsByTagName("ORB"));

                orbNodes.forEach((orbNode, orbIndex) => {

                    const tileId = `${craneId}-tile-${orbIndex + 1}`;

                    const tile = {

                        id: tileId,

                        title: orbNode.querySelector("title")?.textContent.trim() || `Lezione ${orbIndex + 1}`,

                        content: []

                    };

                    Array.from(orbNode.children).forEach(contentNode => {

                        const type = contentNode.tagName.toLowerCase();

                        if (type === 'lesson') {

                            tile.content.push({ type: 'orb', title: "Concetto", text: contentNode.textContent.trim() });

                        } else if (type === 'flashcard') {

                            const question = contentNode.querySelector("question")?.textContent.trim();

                            const answer = contentNode.querySelector("answer")?.textContent.trim();

                            if (question && answer) {

                                tile.content.push({ type: 'flashcard', question, answer });

                            }

                        }

                    });

                    if (tile.content.length > 0) crane.tiles.push(tile);

                });

                if (crane.tiles.length > 0) pathwayData.cranes.push(crane);

            });

            return pathwayData;

        }

        function processContent(xmlContent, sourceName) {

            const pathwayData = parseXML(xmlContent);

           

            if (!pathwayData) {

                showStatus('Formato XML non valido', 'error');

                return;

            }

            const courseId = 'course-' + Date.now();

            loadedCourses.set(courseId, {

                id: courseId,

                title: pathwayData.title,

                source: sourceName,

                data: pathwayData

            });

            renderLoadedCourses();

            saveCoursesToStorage();

            showStatus(`Corso "${pathwayData.title}" caricato!`, 'success');

        }

        function renderLoadedCourses() {

            const container = document.getElementById('loaded-courses');

            container.innerHTML = '';

            if (loadedCourses.size === 0) {

                container.innerHTML = `<p class="text-center text-slate-500 p-4 bg-slate-50 rounded-lg">Nessun corso caricato. Usa le schede in alto per iniziare.</p>`;

                return;

            }

            loadedCourses.forEach(course => {

                const courseElement = document.createElement('div');

                courseElement.className = 'bg-white p-4 rounded-lg shadow-sm border border-slate-200';

                courseElement.innerHTML = `

                    <div class="flex justify-between items-start">

                        <div class="flex-1 mr-4">

                            <h3 class="font-bold text-slate-800">${course.title}</h3>

                            <p class="text-sm text-slate-500">${course.data.cranes.length} capitoli</p>

                        </div>

                        <div class="flex gap-2">

                            <button onclick="startCourse('${course.id}')" class="bg-blue-500 hover:bg-blue-600 text-white px-4 py-2 rounded-lg text-sm font-semibold">Inizia</button>

                            <button onclick="deleteCourse('${course.id}')" class="bg-red-500 hover:bg-red-600 text-white px-3 py-2 rounded-lg text-sm">×</button>

                        </div>

                    </div>

                `;

                container.appendChild(courseElement);

            });

        }

        function startCourse(courseId) {

            const course = loadedCourses.get(courseId);

            if (!course) return;

            currentApp = new LearningApp(course.data);

            showView('pathway');

        }

        function deleteCourse(courseId) {

            loadedCourses.delete(courseId);

            renderLoadedCourses();

            saveCoursesToStorage();

        }

        class LearningApp {

            constructor(pathwayData) {

                this.pathwayData = pathwayData;

                this.state = {

                    progress: { currentCraneIndex: 0, completedTiles: [] },

                    currentTile: null,

                    currentContentIndex: 0,

                    flashcardState: 'question',

                    isAnimating: false

                };

                this.renderPathway();

            }

            renderPathway() {

                document.getElementById('pathway-title').textContent = this.pathwayData.title;

                const pathwayList = document.getElementById('pathway-list');

                pathwayList.innerHTML = '';

                this.pathwayData.cranes.forEach((crane, index) => {

                    const isLocked = index > this.state.progress.currentCraneIndex;

                    const isCompleted = index < this.state.progress.currentCraneIndex;

                    const isActive = index === this.state.progress.currentCraneIndex;

                    const craneEl = document.createElement('div');

                    craneEl.className = `bg-white rounded-2xl shadow-md ${isLocked ? 'opacity-50' : ''}`;

                    let icon = isLocked ? '🔒' : isCompleted ? '✅' : '📖';

                    const header = document.createElement('div');

                    header.className = 'p-5 cursor-pointer flex justify-between items-center';

                    header.innerHTML = `<div><h2 class="text-lg font-bold">${crane.title}</h2></div><div>${icon}</div>`;

                    const tilesContainer = document.createElement('div');

                    tilesContainer.className = 'px-5 pb-5 space-y-3';

                    tilesContainer.style.display = isActive ? 'block' : 'none';

                    crane.tiles.forEach(tile => {

                        const isTileCompleted = this.state.progress.completedTiles.includes(tile.id);

                        const tileEl = document.createElement('div');

                        tileEl.className = `p-4 rounded-lg border cursor-pointer ${isTileCompleted ? 'bg-blue-50 border-blue-200' : 'border-slate-200 hover:bg-slate-50'}`;

                        tileEl.innerHTML = `<p class="font-semibold">${tile.title}</p>`;

                       

                        if (!isLocked) {

                            tileEl.onclick = () => this.startLearningSession(tile);

                        }

                       

                        tilesContainer.appendChild(tileEl);

                    });

                    if (!isLocked) {

                        header.onclick = () => {

                            const open = tilesContainer.style.display === 'none';

                            tilesContainer.style.display = open ? 'block' : 'none';

                        };

                    }

                    craneEl.appendChild(header);

                    craneEl.appendChild(tilesContainer);

                    pathwayList.appendChild(craneEl);

                });

                this.updateGlobalProgress();

            }

                this.renderCurrentContent();

            }

            renderCurrentContent() {

                if (this.state.currentContentIndex >= this.state.currentTile.content.length) {

                    if (!this.state.progress.completedTiles.includes(this.state.currentTile.id)) {

                        this.state.progress.completedTiles.push(this.state.currentTile.id);

                        this.checkForCraneCompletion();

                    }

                    document.getElementById('tile-completion-screen').classList.add('active');

                    return;

                }

                const content = this.state.currentTile.content[this.state.currentContentIndex];

                if (content.type === 'orb') {

                    document.getElementById('orb-title').textContent = content.title;

                    document.getElementById('orb-text').textContent = content.text;

                    document.getElementById('orb-content').classList.add('active');

                        this.renderCurrentContent();

                    };

                } else if (content.type === 'flashcard') {

                    const card = document.getElementById('card');

                    document.getElementById('answer').textContent = content.answer;

                    document.getElementById('flashcard-content').classList.add('active');

                    flipButton.onclick = () => {

                        card.classList.add('is-flipped');

                        flipButton.classList.add('hidden');

                            feedbackButtons.classList.remove('hidden');

                        }, 600);

                    };

                    feedbackButtons.querySelectorAll('.feedback-btn').forEach(b => {

                        b.onclick = () => {

                            this.state.currentContentIndex++;

                        };

                    });

                }

                // Update progress

                const total = this.state.currentTile.content.length;

                const prog = Math.min(this.state.currentContentIndex, total);

                const perc = total > 0 ? Math.round((prog / total) * 100) : 0;

                document.getElementById('progress-bar').style.width = `${perc}%`;

                document.getElementById('session-progress').textContent = `${prog}/${total}`;

            }

            checkForCraneCompletion() {

                const crane = this.pathwayData.cranes[this.state.progress.currentCraneIndex];

                if (!crane) return;

                const allTilesCompleted = crane.tiles.every(tile =>

                    this.state.progress.completedTiles.includes(tile.id)

                );

                if (allTilesCompleted && this.state.progress.currentCraneIndex < this.pathwayData.cranes.length - 1) {

                    this.state.progress.currentCraneIndex++;

                }

            }

            updateGlobalProgress() {

                const total = this.pathwayData.cranes.reduce((s, c) => s + c.tiles.length, 0);

                const perc = total > 0 ? Math.round((this.state.progress.completedTiles.length / total) * 100) : 0;

                document.getElementById('global-progress').style.width = `${perc}%`;

                document.getElementById('progress-text').textContent = `${perc}%`;

            }

Guida Pratica per Creare la Tua App

Segui questi passaggi:

 * Crea una cartella: Sul tuo computer, crea una nuova cartella. Chiamala, ad esempio, la-mia-learning-app.

 * Crea il file index.html:

   * Copia il codice qui sopra.

   * Incollalo in un file di testo e salvalo come index.html all'interno della cartella che hai appena creato.

 * Crea il file manifest.json:

   * Questo file descrive la tua app (nome, icona, colori).

   * Crea un nuovo file di testo nella stessa cartella, incolla il codice qui sotto e salvalo come manifest.json.

   {

  "name": "Learning App",

  "short_name": "LearnApp",

  "start_url": "/index.html",

  "display": "standalone",

  "background_color": "#f1f5f9",

  "theme_color": "#2563eb",

  "description": "La tua app personale per l'apprendimento.",

  "icons": [

    {

      "src": "images/icon-192.png",

      "sizes": "192x192",

      "type": "image/png"

    },

    {

      "src": "images/icon-512.png",

      "sizes": "512x512",

      "type": "image/png"

    }

  ]

}

 * Crea il file sw.js (Service Worker):

   * Questo file gestisce il funzionamento offline.

   * Crea un altro file di testo, incolla il codice qui sotto e salvalo come sw.js nella stessa cartella.

   const CACHE_NAME = 'learning-app-cache-v1';

const urlsToCache = [

  '/',

  '/index.html'

];

self.addEventListener('install', event => {

  event.waitUntil(

    caches.open(CACHE_NAME)

      .then(cache => {

        console.log('Opened cache');

        return cache.addAll(urlsToCache);

      })

  );

});

self.addEventListener('fetch', event => {

  event.respondWith(

    caches.match(event.request)

      .then(response => {

        if (response) {

          return response;

        }

        return fetch(event.request);

      }

    )

  );

});

 * Crea le Icone:

   * Dentro la cartella la-mia-learning-app, crea un'altra cartella chiamata images.

   * Crea due immagini PNG per l'icona della tua app e salvale in questa cartella come icon-192.png (grande 192x192 pixel) e icon-512.png (512x512 pixel). Puoi usare qualsiasi programma di grafica.

 * Carica Online (Hosting):

   * Questo è un passaggio cruciale. Per poter "installare" una PWA, i file devono essere su un server sicuro (HTTPS), non solo sul tuo computer.

   * Puoi usare servizi gratuiti e facilissimi come Netlify o GitHub Pages. Ti basterà trascinare la tua cartella la-mia-learning-app sul loro sito per pubblicarla.

 * Installa sul Cellulare:

   * Una volta che hai caricato i file e hai l'indirizzo web (es. https://la-mia-app.netlify.app), aprilo con il browser del tuo cellulare (Chrome su Android o Safari su iOS).

   * Dovresti vedere un'opzione per "Aggiungi a schermata Home" o "Installa app". Cliccala!

   * Vedrai apparire l'icona della tua app sulla home del telefono. Da quel momento, potrai avviarla come un'app normale, e funzionerà anche se non sei connesso a internet!

Sembra un processo lungo, ma una volta fatto la prima volta è molto semplice. Fammi sapere se hai bisogno di aiuto in uno di questi passaggi!
